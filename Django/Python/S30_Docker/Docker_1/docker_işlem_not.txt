FROM python:alpine3.17
# COPY . /app 
# .(nokta) Bütün dosyaları seçer tek tek almak istersek de aralarına birşey koymadan app.py index.html şeklinde yazabiliriz || /app ile docker için oluşturacağı dizinin adı olacak dizin varsa gideceği nokta olarak nitelendirecek
WORKDIR /apps 
# bu komut ile de çalışma klasçrünü seçeriz yok ise oluşturmasını sağlar
# RUN 
# build aşamasında yapılcak komutlar  
COPY . . 
# kodu ile de WORKDIR ile çalışma dizini belirtildiğinde 1. nokta bulunudğum dizindeki bütün klasör ev dosyaları al 2. nokta ise WORKDIR ile belirtilen koumda aldıklarımla çalış 
CMD python apps.py 
# Buil bittikten sonra çalışacak komutlar
